 70 /**
 71  * The most abstract class for MarkerObject <br> It implements events: <br>
 72  * - 'rightclick' deletes the marker <br>
 73  * - 'drag' changes latitude/longitude at Marker specific fields in form
 74  * @namespace MarkerClass
 75  * @param location usually event coordinates in google map, Marker is created at that specific place
 76  * @param markerProp marker properties <br>
 77  * markerProp[0] = latitude name in form <br>
 78  * markerProp[1] = longitude name in form <br>
 79  * @returns MarkerWithLabel
 80  */
 81 
 82 function AbstractMarkerClass(location, markerProp) {
 83     /**
 84      * Base marker Object
 85      */
 86     var abstractMarkerObj = new MarkerWithLabel({
 87         position: location,
 88         map: mapVariables.map,
 89         draggable: true,
 90         icon: '',
 91         labelContent: containerObj.componentCounter.toString(),
 92         labelClass: "labels",
 93         labelAnchor: new google.maps.Point(14, -5),
 94         latString: markerProp[0],
 95         lngString: markerProp[1]
 96     });
 97     /**
 98      * Removes marker with 'right click'
 99      */
100     var removeMarker = function () {
101         google.maps.event.addListener(abstractMarkerObj, 'rightclick', function () {
102             abstractMarkerObj.setMap(null);
103         });
104     }();
105 
106     /**
107      * Dragging marker updates form coordinates
108      */
109     var dragMarker = function () {
110         google.maps.event.addListener(abstractMarkerObj, 'drag', function (event) {
111             var currentId = returnCurrentId(containerObj.allComponentIdsArray);
112             var currentMarkerId = parseInt(abstractMarkerObj.labelContent);
113             if (currentId === currentMarkerId) {
114                 document.getElementById(abstractMarkerObj.latString).value = event.latLng.lat();
115                 document.getElementById(abstractMarkerObj.lngString).value = event.latLng.lng();
116             }
117         });
118     }();
119 
120     /**
121      * Puts latitude/longitude to form
122      */
123     var putLatLngToForm = function () {
124         document.getElementById(abstractMarkerObj.latString).value = abstractMarkerObj.position.lat();
125         document.getElementById(abstractMarkerObj.lngString).value = abstractMarkerObj.position.lng();
126     }();
127     /**
128      * Returs current id during filling the form. <br>
129      * It looks for allComponentIdsArray element which value in form is different than NaN and returns it
130      * @param allComponentIdsArray is all form ids like 'quest_id', 'info_id'
131      * @returns id of current form
132      */
133     function returnCurrentId(allComponentIdsArray) {
134         for (var i = 0; i < allComponentIdsArray.length; i++) {
135             var potentialCurrentId = parseInt(document.getElementById(allComponentIdsArray[i]).value);
136             if (!isNaN(potentialCurrentId)) {
137                 return potentialCurrentId;
138             }
139         }
140     }
141     alert(markerProp.join('\n'))
142     return abstractMarkerObj;
143 }

145 /**
146  * Abstract Class that represents Component Marker
147  * It adds properties like: <br>
148  * markerProp[2] = id of id in form (quest_id or info_id)<br>
149  * markerProp[3] = id of form <br>
150  * @namespace MarkerClass
151  * @param location
152  * @param markerProp
153  * @extends AbstractMarkerClass
154  * @returns Component Marker
155  */
156 function ComponentMarkerClass(location, markerProp) {
157     containerObj.componentCounter++;
158     /**
159      * Creates AbstractMarker and adds new properties to it
160      */
161     var markerObj = function () {
162         var concreteMarkerObj = AbstractMarkerClass(location, markerProp);
163         concreteMarkerObj.queryString = 0;
164         concreteMarkerObj.idString = markerProp[2];
165         concreteMarkerObj.formString = markerProp[3];
166         concreteMarkerObj.formId = 0;
167         return concreteMarkerObj;
168     }();
169     /**
170      * Removes Component Marker after 'right click' event
171      */
172     var removeMarker = function () {
173         google.maps.event.addListener(markerObj, 'rightclick', function () {
174             var componentIndexToRemove = parseInt(markerObj.labelContent);
175             containerObj.queryStringsArr.splice(componentIndexToRemove, 1);
176             containerObj.componentCounter--;
177             MenuManager.goToChooseComponentTab();
178             MenuManager.cleanForms();
179         });
180     }();
181     /**
182      * Handles Component Marker click event
183      */
184 
185     google.maps.event.addListener(markerObj, 'click', function (event) {
186         if (markerObj.queryString !== 0) {
187             var $form = $("#".concat(markerObj.formString));
188             var formSerializedData = markerObj.queryString;
189             MenuManager.cleanForms();
190             $form.deserialize(formSerializedData);
191             MenuManager.goToConcreteForm(markerObj.formID);
192             $(".radio_checked_true").each(function () {
193                 if ($(this).is(':checked')) {
194                     var elem = $("ol.formset", $(this).parent());
195                     elem.slideDown('fast');
196                 }
197             });
198         } else {
199             putComponentIdToForm();
200             MenuManager.goToConcreteForm(markerObj.formID);
201         }
202     });
203 
204 
205     /**
206      * Puts Component Id to concrete form (quest_id or info_id)
207      */
208     var putComponentIdToForm = function () {
209         document.getElementById(markerObj.idString).value = containerObj.componentCounter;
210     }();
211 
212     containerObj.componentArray.push(markerObj);
213     return markerObj;
214 }

217 /**
218  * Class that represents Circle that will be bind to TaskMarker
219  * @namespace Circle class
220  */
221 function CircleClass(taskPosition) {
222     var circleObj = new google.maps.Circle({
223         center: taskPosition,
224         radius: 0,
225         circleId: containerObj.componentCounter,
226         strokeColor: "#0000FF",
227         strokeOpacity: 0.8,
228         strokeWeight: 2,
229         fillColor: "#0000FF",
230         fillOpacity: 0.4,
231         map: mapVariables.map
232     });
233     return circleObj;
234 }
235 /**
236  * Class that represents Task Marker at given position <br>
237  * adds icon property to Abstract Marker
238  * @namespace TaskMarkerClass
239  * @param location of Task Marker
240  * @returns Task Marker

255     }();
256 
257     containerObj.taskArray.push(taskMarkerObj);

272 /**
273  * Class that represents Marker with Circle binded to id
274  * @namespace MarkerWithCircleClass
275  * @param location location of Marker/Circle
276  * @param idString id of id in form
277  * @param radiusString id of radius in form
278  */
279 function MarkerWithCircleClass(location, idString, radiusString) {
280     var taskMarker = TaskMarkerClass(location);
281     var taskPosition = taskMarker.position;
282     var taskCircle = TaskCircleClass(taskPosition);
283 
284     /**
285      * Handles changing radius in form <br>
286      * User sees changes immediately on the map
287      */
288     var changeRadiusHandler = function () {
289         google.maps.event.addDomListener(
290         $("#".concat(radiusString)).bind('input', function () {
291             var currentTaskId = parseInt(document.getElementById(idString).value);
292             for (var i = 0; i < containerObj.taskCircleArray.length; i++) {
293                 if (currentTaskId === (containerObj.taskCircleArray[i]["circleId"])) {
294                     containerObj.taskCircleArray[i].bindTo('center', containerObj.taskArray[currentTaskId], 'position');
295                     containerObj.taskCircleArray[i].setRadius(parseInt(document.getElementById(radiusString).value));
296                 }
297             }
298         }));
299     }();
300     /**
301      * Removes marker with circle after 'right click' event
302      */
303     var removeMarkerWithCircle = function () {
304         google.maps.event.addListener(taskMarker, 'rightclick', function () {
305             taskCircle.setMap(null);
306             document.getElementById(radiusString).value = null;
307         });
308     }();
309 }
